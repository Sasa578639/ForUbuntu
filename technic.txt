ВЕБ_интерфейс_2:
sudo systemctl stop nginx
sudo rm -f /etc/nginx/sites-enabled/technic
sudo rm -f /etc/nginx/sites-available/technic

sudo tee /etc/nginx/sites-available/technic > /dev/null <<'EOF'
server {
    listen 80;
    server_name _;
    
    root /home/aleks/.ros/www;
    index index.html index.htm;
    
    # Включаем поддержку символических ссылок
    disable_symlinks off;
    
    location / {
        autoindex on;
        autoindex_exact_size off;
        autoindex_localtime on;
        try_files $uri $uri/ =404;
    }
    
    # Разрешаем доступ к папкам с файлами
    location ~* \.(html|css|js|png|jpg|jpeg|gif|svg|ico|json|xml|txt)$ {
        expires 1d;
        add_header Cache-Control "public, immutable";
    }
}
EOF
